%Use a loop to create volatege and time
%create summation variable
V = 0;
%loop counter variable
t=0;
%loop
for t = 0:1:400
    V(t+1) =3^(7-t/100);
    time(t+1)=t/10;

end
%Plot Data
figure
plot(time,V)
xlabel('Time (s)')
ylabel('Voltage (V)')
title('Capacitor Dissipation Linear')

%Plot with semilogx
figure
semilogx(time,V,'o')
xlabel('log Time (s)')
ylabel('Voltage (V)')
title('Capacitor Dissipation log(Time)')

%Plot with semilogy
figure
semilogy(time,V,'-')
xlabel('Time (s)')
ylabel('log Voltage (V)')
title('Capacitor Dissipation log(Voltage)')

%Plot with loglog
figure
loglog(time,V,'*')
xlabel('log Time (s)')
ylabel('log Voltage (V)')
title('Capacitor Dissipation log(Voltage) log(Time)')
